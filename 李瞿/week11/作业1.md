1. 数据管理接口
# 上传文档到知识库
POST /upload/document
{
    "knowledge_base_id": "string",
    "file": "file object (PDF/TXT/DOCX)"
}

# 获取所有文档列表
GET /documents/list
{
    "knowledge_base_id": "string"
}

# 删除文档
DELETE /document/{doc_id}

2. 多模态检索接口
# 检索相关图文内容（支持文本+图片查询）
POST /search/multimodal
{
    "query_text": "string",          // 文本查询
    "query_image": "base64 string",   // 图片查询（可选）
    "top_k": 10,                     // 返回数量
    "knowledge_base_id": "string"    // 知识库ID
}

3. 多模态问答接口
# 提问并获取答案（结合检索结果）
POST /chat
{
    "query": "string",               // 用户问题
    "knowledge_base_id": "string",   // 知识库ID
    "context_docs": ["doc_id"]       // 可选：指定上下文文档
}
返回格式：
{
    "status": 200,
    "message": "success",
    "data": {
        "answer": "string",
        "sources": [
            {
                "doc_id": "string",
                "page": int,
                "chunk_id": "string",
                "content": "string",
                "image_url": "string"  // 如果有相关图片
            }
        ]
    }
}
